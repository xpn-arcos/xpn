41int8_t DARRAY_InsEndDarray(/*INOUT*/ t_pointerDArray *t,
42 /*INOUT*/ long *n,
43 /*IN*/ T_POINTER gptr)
44{
45 /* check params */
46 if (NULL == n)
47 return (-1);
48
49 /* Insert into the End of Darray */
50 if ((*t) == (t_pointerDArray)NULL)
51 {
52 (*t) = (t_pointerDArray)malloc(2 * c_POINTER);
53 if (NULL == (*t))
54 return (-1);
55
56 (*n) = 1;
57 (*t)[0] = gptr;
58 (*t)[1] = (T_POINTER)NULL;
59 }
60 else /* ( (*t) != (t_pointerDArray)NULL ) */
61 {
62 t_pointerDArray taux;
63
64 (*n)++;
65 taux = (t_pointerDArray)realloc((*t), ((*n) + 1) * c_POINTER);
66 if (NULL == taux)
67 return (-1);
68
69 (*t) = taux;
70 taux[(*n)] = (T_POINTER)NULL;
71 taux[(*n) - 1] = gptr;
72 }
73
74 /* return ok */
75 return (1);
76}

280int8_t DARRAY_DelEltoDarray(/*INOUT*/ t_pointerDArray *t,
281 /*IN*/ T_POINTER elto)
282{
283 long c, pentry;
284
285 /* check params */
286 if (NULL == (*t))
287 return (-1);
288 if (NULL == elto)
289 return (-1);
290
291 /* DelEltoDarray */
292 for (c = 0; ((*t)[c] && ((*t)[c] != elto)); c++)
293 ;
294 if (((*t)[c]) == (t_pointerDArray)NULL)
295 {
296 return (-1);
297 }
298 pentry = c;
299 c++;
300 for (; ((*t)[c] != NULL); c++)
301 ;
302 memmove(&((*t)[pentry]),
303 &((*t)[pentry + 1]),
304 (c - pentry) * c_POINTER);
305 (*t) = (t_pointerDArray)realloc(*t, c * c_POINTER);
306 if (c == 1)
307 {
308 free(*t);
309 (*t) = NULL;
310 /* there was only a element that was deleted... and NULL, of course */
311 }
312
313 /* return ok */
314 return (1);
315}

351/**
352 * Return the 'orden'-th element in the 't' dynamic array,
353 * in a NULL ended array.
354 * @param t a dynamic array.
355 * @param orden position (begin at 0) of element to return.
356 * @return the element at this position or NULL.
357 */
358T_POINTER DARRAY_GetNFromDarray2(/*IN*/ t_pointerDArray t,
359 /*IN*/ long orden)
360{
361 long i;
362
363 /* check params */
364 if (NULL == t)
365 return (T_POINTER)NULL;
366
367 /* searching... */
368 for (i = 0; ((t[i] != NULL) && (i != orden)); i++)
369 ;
370 if (i != orden)
371 return (T_POINTER)NULL;
372
373 /* return element */
374 return (t[orden]);
375}
