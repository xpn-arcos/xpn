55void NFItoNFSattr(fattr *nfs_att, struct nfi_attr *nfi_att)
56{
57 switch (nfi_att->at_type)
58 {
59 case NFIDIR:
60 nfs_att->type = NFDIR;
61 break;
62 case NFIFILE:
63 nfs_att->type = NFREG;
64 break;
65 }
66
67 nfs_att->mode = (u_long)nfi_att->at_mode;
68 /* u_long nlink; */
69 nfs_att->uid = (u_long)nfi_att->at_uid;
70 nfs_att->gid = (u_long)nfi_att->at_gid;
71 nfs_att->size = (u_long)nfi_att->at_size;
72 nfs_att->blocksize = (u_long)nfi_att->at_blksize;
73 /* u_long rdev; */
74 nfs_att->blocks = (u_long)nfi_att->at_blocks;
75 /* u_long fsid; */
76 /* u_long fileid;*/
77 nfs_att->atime.seconds = (u_long)nfi_att->at_atime;
78 nfs_att->atime.useconds = (u_long)0;
79 nfs_att->mtime.seconds = (u_long)nfi_att->at_atime;
80 nfs_att->mtime.useconds = (u_long)0;
81 nfs_att->ctime.seconds = (u_long)nfi_att->at_ctime;
82 nfs_att->ctime.useconds = (u_long)0;
83}

85void NFStoNFIattr(struct nfi_attr *nfi_att, fattr *nfs_att)
86{
87
88 switch (nfs_att->type)
89 {
90 case NFDIR:
91 nfi_att->at_type = NFIDIR;
92 break;
93 case NFREG:
94 nfi_att->at_type = NFIFILE;
95 break;
96 default:
97 break;
98 }
99
100 nfi_att->at_mode = (mode_t)nfs_att->mode;
101 /* u_long nlink; */
102 nfi_att->at_uid = (uid_t)nfs_att->uid;
103 nfi_att->at_gid = (gid_t)nfs_att->gid;
104 nfi_att->at_size = (off_t)nfs_att->size;
105 nfi_att->at_blksize = (u_long)nfs_att->blocksize;
106 /* u_long rdev; */
107 nfi_att->at_blocks = (u_long)nfs_att->blocks;
108 /* u_long fsid; */
109 /* u_long fileid;*/
110 nfi_att->at_atime = (time_t)nfs_att->atime.seconds + (1000 * nfs_att->atime.useconds);
111 nfi_att->at_mtime = (time_t)nfs_att->mtime.seconds + (1000 * nfs_att->mtime.useconds);
112 nfi_att->at_ctime = (time_t)nfs_att->ctime.seconds + (1000 * nfs_att->ctime.useconds);
113
114 nfi_att->private_info = NULL;
115}

128int nfi_nfs_init(char *url, struct nfi_server *serv, __attribute__((__unused__)) struct nfi_attr_server *attr)
129{
130 char server[NFSPATHLEN], dir[NFSPATHLEN], prt[NFSPATHLEN];
131 int ret;
132 struct nfi_nfs_server *server_aux;
133
134 if (serv == NULL)
135 {
136 nfs_err(NFSERR_PARAM);
137 fprintf(stderr, "nfi_nfs_init: serv is null. Don't inicialize this variable\n");
138 return -1;
139 }
140
141 /* functions */
142 serv->ops = (struct nfi_ops *)malloc(sizeof(struct nfi_ops));
143 if (serv->ops == NULL)
144 {
145 nfs_err(NFSERR_MEMORY);
146 fprintf(stderr, "nfi_nfs_init: serv->ops is null. Can't reserve memory\n");
147 return -1;
148 }
149
150 serv->ops->nfi_reconnect = nfi_nfs_reconnect;
151 serv->ops->nfi_disconnect = nfi_nfs_disconnect;
152
153 serv->ops->nfi_preload = NULL;
154 serv->ops->nfi_flush = NULL;
155
156 serv->ops->nfi_getattr = nfi_nfs_getattr;
157 serv->ops->nfi_setattr = nfi_nfs_setattr;
158
159 serv->ops->nfi_open = nfi_nfs_open;
160 serv->ops->nfi_close = nfi_nfs_close;
161
162 serv->ops->nfi_read = nfi_nfs_read;
163 serv->ops->nfi_write = nfi_nfs_write;
164
165 serv->ops->nfi_create = nfi_nfs_create;
166 serv->ops->nfi_remove = nfi_nfs_remove;
167 serv->ops->nfi_rename = nfi_nfs_rename;
168
169 serv->ops->nfi_mkdir = nfi_nfs_mkdir;
170 serv->ops->nfi_rmdir = nfi_nfs_rmdir;
171 serv->ops->nfi_opendir = nfi_nfs_opendir;
172 serv->ops->nfi_readdir = nfi_nfs_readdir;
173 serv->ops->nfi_closedir = nfi_nfs_closedir;
174 serv->ops->nfi_statfs = nfi_nfs_statfs;
175
176 ret = ParseURL(url, prt, NULL, NULL, server, NULL, dir);
177 if (ret < 0)
178 {
179 nfs_err(NFSERR_URL);
180 free(serv->ops);
181 return -1;
182 }
183
184 server_aux = (struct nfi_nfs_server *)malloc(sizeof(struct nfi_nfs_server));
185 if (server_aux == NULL)
186 {
187 nfs_err(NFSERR_MEMORY);
188 fprintf(stderr, "nfi_nfs_init: url %s incorrect.\n", url);
189 return -1;
190 }
191 server_aux->cl = (CLIENT *)create_connection_mount(server, NFS_UDP);
192
193 if (server_aux->cl == NULL)
194 {
195 nfs_err(NFSERR_MNTCONNECTION);
196 fprintf(stderr, "nfi_nfs_init: server_aux is null. Can't reserve memory\n");
197 free(server_aux);
198 return -1;
199 }
200 ret = nfs_mount(dir, server_aux->fh, server_aux->cl);
201#if defined(DEBUG_NFI)
202 printf("desp. de montar %d\n", ret);
203#endif
204 if (ret < 0)
205 {
206 nfs_err(NFSERR_MOUNT);
207 close_connection_mount(server_aux->cl);
208 fprintf(stderr, "nfi_nfs_init: Can't mount %s in server %s (err:%d).\n", dir, server, ret);
209 free(server_aux);
210 return -1;
211 }
212
213 close_connection_mount(server_aux->cl);
214 /*
215 if((strcmp(prt, "nfs2tcp") == 0) ||
216 (strcmp(prt, "nfs2tcphilos") == 0)){
217 #if defined(DEBUG_NFI)
218 printf("nfs2tcp\n");
219 #endif
220 */
221 server_aux->cl = create_connection_nfs(server, NFS_TCP);
222 /* }
223 else{
224 server_aux->cl = create_connection_nfs(server, NFS_UDP);
225 }
226 */
227 if (server_aux->cl == NULL)
228 {
229 nfs_err(NFSERR_NFSCONNECTION);
230 fprintf(stderr, "nfi_nfs_init: Can't connect with NFS service of %s\n", server);
231 free(server_aux);
232 return -1;
233 }
234
235 serv->private_info = (void *)server_aux;
236
237 // serv->protocol = NFS;
238
239 serv->server = (char *)malloc(sizeof(char) * (strlen(server) + 1)); /* server address */
240 if (serv->server == NULL)
241 {
242 free(server_aux);
243 nfs_err(NFSERR_MEMORY);
244 fprintf(stderr, "nfi_nfs_init: serv->server is null. Can't reserve memory\n");
245 return -1;
246 }
247 strcpy(serv->server, server);
248
249 serv->url = (char *)malloc(sizeof(char) * (strlen(url) + 1)); /* server address */
250 if (serv->url == NULL)
251 {
252 nfs_err(NFSERR_MEMORY);
253 close_connection_nfs(server_aux->cl);
254 fprintf(stderr, "nfi_nfs_init: serv->url is null. Can't reserve memory\n");
255 free(serv->server);
256 free(server_aux);
257 return -1;
258 }
259 strcpy(serv->url, url);
260 serv->wrk = (struct nfi_worker *)malloc(sizeof(struct nfi_worker));
261 memset(serv->wrk, 0, sizeof(struct nfi_worker));
262 if (serv->wrk == NULL)
263 {
264 close_connection_nfs(server_aux->cl);
265 free(serv->server);
266 fprintf(stderr, "nfi_nfs_init: serv->wrk is null. Can't reserve memory\n");
267 free(serv->url);
268 free(server_aux);
269 return -1;
270 }
271 if ((strcmp(prt, "nfshilos") == 0) ||
272 (strcmp(prt, "nfs2hilos") == 0) ||
273 (strcmp(prt, "nfs2udphilos") == 0) ||
274 (strcmp(prt, "nfs2tcphilos") == 0))
275 {
276#if defined(DEBUG_NFI)
277 printf("nfs2hilos\n");
278#endif
279 nfi_worker_init(serv->wrk, serv, 1);
280 }
281 else
282 {
283 nfi_worker_init(serv->wrk, serv, 0);
284 }
285 return 0;
286}

288/************************************************************

291int nfi_nfs_disconnect(struct nfi_server *serv)
292{
293
294 struct nfi_nfs_server *server_aux;
295 char dir[NFSPATHLEN];
296 int ret;
297
298 if (serv == NULL)
299 return 0;
300
301 server_aux = (struct nfi_nfs_server *)serv->private_info;
302 if (server_aux != NULL)
303 {
304 if (server_aux->cl != NULL)
305 close_connection_nfs(server_aux->cl);
306
307 ret = ParseURL(serv->url, NULL, NULL, NULL, NULL, NULL, dir);
308 if (ret < 0)
309 {
310 nfs_err(NFSERR_URL);
311 fprintf(stderr, "nfi_nfs_disconnect: url %s incorrect.\n", serv->url);
312 free(serv->private_info);
313 return -1;
314 }
315 server_aux->cl = create_connection_mount(serv->server, NFS_UDP);
316 if (server_aux->cl != NULL)
317 {
318 ret = nfs_umount(dir, server_aux->cl);
319 close_connection_mount(server_aux->cl);
320 }
321
322 free(serv->private_info);
323 }
324
325 if (serv->url != NULL)
326 {
327 free(serv->url);
328 }
329
330 if (serv->server != NULL)
331 {
332 free(serv->server);
333 }
334
335 // serv->protocol = -1;
336
337 nfi_worker_end(serv->wrk);
338
339 return 0;

597int nfs_open(struct nfi_server *serv, char *url, struct nfi_fhandle *fho)
598{
599 char dir[NFSPATHLEN], server[NFSPATHLEN];
600 int ret;
601 struct nfi_nfs_server *server_aux;
602 struct nfi_nfs_fhandle *fh_aux;
603
604 if (serv == NULL)
605 {
606 nfs_err(NFSERR_PARAM);
607 return -1;
608 }
609
610 if (fho == NULL)
611 {
612 nfs_err(NFSERR_PARAM);
613 return -1;
614 }
615
616#ifdef NFI_DYNAMIC
617 if (serv->private_info == NULL)
618 {
619 ret = nfi_nfs_reconnect(serv);
620 if (ret < 0)
621 {
622 /* nfs_err(); not necessary */
623 return -1;
624 }
625 }
626#else
627 if (serv->private_info == NULL)
628 {
629 nfs_err(NFSERR_PARAM);
630 return -1;
631 }
632#endif
633
634 ret = ParseURL(url, NULL, NULL, NULL, server, NULL, dir);
635 if (ret < 0)
636 {
637 nfs_err(NFSERR_URL);
638 fprintf(stderr, "nfs_open: url %s incorrect.\n", url);
639 return -1;
640 }
641
642 fho->url = (char *)malloc(strlen(url) + 1);
643 if (fho->url == NULL)
644 {
645 nfs_err(NFSERR_MEMORY);
646 return -1;
647 }
648
649 strcpy(fho->url, url);
650
651 fh_aux = (struct nfi_nfs_fhandle *)malloc(sizeof(struct nfi_nfs_fhandle));
652 if (fh_aux == NULL)
653 {
654 nfs_err(NFSERR_MEMORY);
655 fprintf(stderr, "nfs_open: fho->url is null. Can't reserve memory\n");
656 free(fho->url);
657 return -1;
658 }
659
660 server_aux = (struct nfi_nfs_server *)serv->private_info;
661
662 getDirWithURL(serv->url, dir);
663
664 /* obtaine the NFS file handle */
665 ret = nfs_lookup(server_aux->fh,
666 dir,
667 fh_aux->fh,
668 NULL,
669 server_aux->cl);
670
671#define DBG_NFI 0
672#ifdef DBG_NFI
673 // fprintf(stderr,"nfs_open: lookup %s in server %s (err:%d).\n",dir,serv->server,ret);
674#endif
675
676 if (ret < 0)
677 {
678 // fprintf(stderr,"nfs_open: Fail lookup %s in server %s (err:%d).\n",dir,serv->server,ret);
679 // nfs_err(NFSERR_LOOKUP);
680 // free(fho->url);
681 free(fh_aux);
682 return -1;
683 }
684
685 switch (ret)
686 {
687 case NFREG:
688 fho->type = NFIFILE;
689 break;
690 case NFDIR:
691 fho->type = NFIDIR;
692 break;
693 }
694 fho->server = NULL;
695 fho->priv_fh = NULL;
696 fho->server = serv;
697 fho->priv_fh = (void *)fh_aux;
698
699 return 0;
700}

720int nfi_nfs_close(struct nfi_server *server, struct nfi_fhandle *fh)
721{
722 if (server == NULL)
723 {
724 nfs_err(NFSERR_PARAM);
725 fprintf(stderr, "nfi_nfs_close: server is null.\n");
726 return -1;
727 }
728 if (fh == NULL)
729 {
730 nfs_err(NFSERR_PARAM);
731 fprintf(stderr, "nfi_nfs_close: fh is null.\n");
732 return -1;
733 }
734
735 if (fh->priv_fh != NULL)
736 {
737 /* free memory */
738 // fprintf(stderr,"nfi_nfs_close: fh->priv_fh is null.\n");
739 free(fh->priv_fh);
740 fh->priv_fh = NULL;
741 }
742
743 if (fh->url != NULL)
744 {
745 free(fh->url);
746 }
747
