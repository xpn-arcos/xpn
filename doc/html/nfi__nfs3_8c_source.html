40void NFItoNFS3attr(fattr3 *nfs3_att, struct nfi_attr *nfi_att)
41{
42 nfs3_att->st_dev = nfi_att->st_dev;
43 nfs3_att->st_ino = nfi_att->st_ino;
44
45 switch (nfi_att->at_type)
46 {
47 case NFIDIR:
48 nfs3_att->type = NF3DIR;
49 break;
50 case NFIFILE:
51 nfs3_att->type = NF3REG;
52 break;
53 }
54
55 nfs3_att->mode = (u_long)nfi_att->at_mode;
56 /* u_long nlink; */
57 nfs3_att->uid = (u_long)nfi_att->at_uid;
58 nfs3_att->gid = (u_long)nfi_att->at_gid;
59 nfs3_att->size = (u_long)nfi_att->at_size;
60 // nfs3_att->blocksize = (u_long)nfi_att->at_blksize;
61 /* u_long rdev; */
62 // nfs3_att->blocks = (u_long)nfi_att->at_blocks;
63 /* u_long fsid; */
64 /* u_long fileid;*/
65 nfs3_att->atime.seconds = (u_long)nfi_att->at_atime;
66 // nfs3_att->atime.useconds = (u_long)0;
67 nfs3_att->mtime.seconds = (u_long)nfi_att->at_atime;
68 // nfs3_att->mtime.useconds = (u_long)0;
69 nfs3_att->ctime.seconds = (u_long)nfi_att->at_ctime;
70 // nfs3_att->ctime.useconds = (u_long)0;
71}

84 nfi_att->at_type = NFIFILE;

88 }
89
90 nfi_att->at_mode = (mode_t)nfs3_att->mode;
91 /* u_long nlink; */

93 nfi_att->at_gid = (gid_t)nfs3_att->gid;
94 nfi_att->at_size = (off_t)nfs3_att->size;
95 // nfi_att->at_blksize = (u_long)nfs3_att->blocksize;
96 /* u_long rdev; */
97 // nfi_att->at_blocks = (u_long)nfs3_att->blocks;
98 /* u_long fsid; */
99 /* u_long fileid;*/
100 nfi_att->at_atime = (time_t)nfs3_att->atime.seconds;
101 //+ (1000*nfs3_att->atime.useconds);
102 nfi_att->at_mtime = (time_t)nfs3_att->mtime.seconds;
103 //+ (1000*nfs3_att->mtime.useconds);

106
107 nfi_att->private_info = NULL;
108}

122int nfi_nfs3_init(char *url, struct nfi_server *serv, __attribute__((__unused__)) struct nfi_attr_server *attr)
123{
124 char server[MNTNAMLEN], dir[MNTNAMLEN], prt[MNTNAMLEN];
125 int ret;
126 struct nfi_nfs3_server *server_aux;
127
128 if (serv == NULL)
129 {
130 nfs3_err(NFS3ERR_PARAM);
131 fprintf(stderr, "nfi_nfs3_init: serv is null. Don't inicialize this variable\n");
132 return -1;
133 }
134 /* functions */
135 serv->ops = (struct nfi_ops *)malloc(sizeof(struct nfi_ops));
136 if (serv->ops == NULL)
137 {
138 nfs3_err(NFS3ERR_MEMORY);
139 fprintf(stderr, "nfi_nfs3_init: serv->ops is null. Can't reserve memory\n");
140 return -1;
141 }
142
143 serv->ops->nfi_reconnect = nfi_nfs3_reconnect;
144 serv->ops->nfi_disconnect = nfi_nfs3_disconnect;
145
146 serv->ops->nfi_getattr = nfi_nfs3_getattr;
147 serv->ops->nfi_setattr = nfi_nfs3_setattr;
148
149 serv->ops->nfi_open = nfi_nfs3_open;
150 serv->ops->nfi_close = nfi_nfs3_close;
151
152 serv->ops->nfi_read = nfi_nfs3_read;
153 serv->ops->nfi_write = nfi_nfs3_write;
154
155 serv->ops->nfi_create = nfi_nfs3_create;
156 serv->ops->nfi_remove = nfi_nfs3_remove;
157 serv->ops->nfi_rename = nfi_nfs3_rename;
158
159 serv->ops->nfi_mkdir = nfi_nfs3_mkdir;
160 serv->ops->nfi_rmdir = nfi_nfs3_rmdir;
161 serv->ops->nfi_opendir = nfi_nfs3_opendir;
162 serv->ops->nfi_readdir = nfi_nfs3_readdir;
163 serv->ops->nfi_closedir = nfi_nfs3_closedir;
164 serv->ops->nfi_statfs = nfi_nfs3_statfs;
165
166 // printf("url %s\n",url);
167 ret = ParseURL(url, prt, NULL, NULL, server, NULL, dir);
168 if (ret < 0)
169 {
170 nfs3_err(NFS3ERR_URL);
171 fprintf(stderr, "nfi_nfs3_init: url %s incorrect.\n", url);
172 free(serv->ops);
173 return -1;
174 }
175
176 // printf("prt %s server %s dir %s\n",prt,server,dir);
177 server_aux = (struct nfi_nfs3_server *)malloc(sizeof(struct nfi_nfs3_server));
178 if (server_aux == NULL)
179 {
180 nfs3_err(NFS3ERR_MEMORY);
181 fprintf(stderr, "nfi_nfs3_init: server_aux is null. Can't reserve memory\n");
182 return -1;
183 }
184 server_aux->cl = (CLIENT *)create_connection_mount3(server, NFS3_UDP);
185
186 if (server_aux->cl == NULL)
187 {
188 nfs3_err(NFS3ERR_MNTCONNECTION);
189 fprintf(stderr, "nfi_nfs3_init: Can't connect with MNT service of %s\n", server);
190 free(server_aux);
191 return -1;
192 }
193 // printf("antes de mount(%s)\n",dir);
194 ret = nfs3_mount(dir, &server_aux->fh, server_aux->cl);
195 // printf("desp. de mount(%s) %d\n",dir,ret);
196#if defined(DEBUG_NFI)
197 printf("desp. de montar %d\n", ret);
198#endif
199 if (ret < 0)
200 {
201 nfs3_err(-ret);
202 close_connection_mount3(server_aux->cl);
203 fprintf(stderr, "nfi_nfs3_init: Can't mount %s in server %s (err:%d).\n", dir, server, ret);
204 free(server_aux);
205 return -1;
206 }
207 nfs3_err(ret);
208
209 close_connection_mount3(server_aux->cl);
210 if ((strcmp(prt, "nfs3") == 0) ||
211 (strcmp(prt, "nfs3tcp") == 0) ||
212 (strcmp(prt, "nfs3tcphilos") == 0))
213 {
214#if defined(DEBUG_NFI)
215 printf("nfs3\n");
216#endif
217
218 // printf("nfs3tcp\n");
219 server_aux->cl = create_connection_nfs3(server, NFS3_TCP);
220 }
221 else
222 {
223 // printf("nfs3\n");
224 server_aux->cl = create_connection_nfs3(server, NFS3_UDP);
225 }
226 if (server_aux->cl == NULL)
227 {
228 nfs3_err(NFS3ERR_NFSCONNECTION);
229 fprintf(stderr, "nfi_nfs3_init: Can't connect with NFS service of %s\n", server);
230 free(server_aux);
231 return -1;
232 }
233
234 serv->private_info = (void *)server_aux;
235
236 // serv->protocol = NFS3;
237
238 serv->server = (char *)malloc(sizeof(char) * (strlen(server) + 1)); /* server address */
239 if (serv->server == NULL)
240 {
241 fprintf(stderr, "nfi_nfs3_init: serv->server is null. Can't reserve memory\n");
242 nfs3_err(NFS3ERR_MEMORY);
243 return -1;
244 }
245 strcpy(serv->server, server);
246
247 serv->url = (char *)malloc(sizeof(char) * (strlen(url) + 1)); /* server address */
248 if (serv->url == NULL)
249 {
250 fprintf(stderr, "nfi_nfs3_init: serv->url is null. Can't reserve memory\n");
251 nfs3_err(NFS3ERR_MEMORY);
252 close_connection_nfs3(server_aux->cl);
253 free(serv->server);
254 free(server_aux);
255 return -1;
256 }
257 strcpy(serv->url, url);
258 serv->wrk = (struct nfi_worker *)malloc(sizeof(struct nfi_worker));
259 memset(serv->wrk, 0, sizeof(struct nfi_worker));
260 if (serv->wrk == NULL)
261 {
262 fprintf(stderr, "nfi_nfs3_init: serv->wrk is null. Can't reserve memory\n");
263 close_connection_nfs3(server_aux->cl);
264 free(serv->server);
265 free(serv->url);
266 free(server_aux);
267 return -1;
268 }
269 if ((strcmp(prt, "nfs3hilos") == 0) ||
270 (strcmp(prt, "nfs3udphilos") == 0) ||
271 (strcmp(prt, "nfs3tcphilos") == 0))
272 {
273#if defined(DEBUG_NFI)
274 printf("nfs3hilos\n");
275#endif
276 nfi_worker_init(serv->wrk, serv, 1);
277 }
278 else
279 {
280 nfi_worker_init(serv->wrk, serv, 0);
281 }
282 return 0;
283}

285/************************************************************
286 * Disconnect to the server *
287 * **********************************************************/
288int nfi_nfs3_disconnect(struct nfi_server *serv)
289{
290
291 struct nfi_nfs3_server *server_aux;
292 char dir[MNTNAMLEN];
293 int ret;
294
295 if (serv == NULL)
296 return 0;
297
298 server_aux = (struct nfi_nfs3_server *)serv->private_info;
299 if (server_aux != NULL)
300 {
301 if (server_aux->cl != NULL)
302 close_connection_nfs3(server_aux->cl);
303
304 ret = ParseURL(serv->url, NULL, NULL, NULL, NULL, NULL, dir);
305 if (ret < 0)
306 {
307 fprintf(stderr, "nfi_nfs3_disconnect: url %s incorrect.\n", serv->url);
308 nfs3_err(NFS3ERR_URL);
309 free(serv->private_info);
310 return -1;
311 }
312 server_aux->cl = create_connection_mount3(serv->server, NFS3_UDP);
313 if (server_aux->cl != NULL)
314 {
315 ret = nfs3_umount(dir, server_aux->cl);
316 close_connection_mount3(server_aux->cl);
317 }
318
319 free(serv->private_info);
320 }
321
322 if (serv->url != NULL)
323 {
324 free(serv->url);
325 }
326
327 if (serv->server != NULL)
328 {
329 free(serv->server);
330 }
331
332 // serv->protocol = -1;
333
334 nfi_worker_end(serv->wrk);
335
336 return 0;
337}

596int nfs3_open(struct nfi_server *serv, char *url, struct nfi_fhandle *fho)
597{
598 char dir[MNTNAMLEN], server[MNTNAMLEN];
599 int ret;
600 struct nfi_nfs3_server *server_aux;
601 struct nfi_nfs3_fhandle *fh_aux;
602
603 if (serv == NULL)
604 {
605 nfs3_err(NFS3ERR_PARAM);
606 return -1;
607 }
608
609 if (fho == NULL)
610 {
611 nfs3_err(NFS3ERR_PARAM);
612 return -1;
613 }
614
615#ifdef NFI_DYNAMIC
616 if (serv->private_info == NULL)
617 {
618 ret = nfi_nfs3_reconnect(serv);
619 if (ret < 0)
620 {
621 /* nfs3_err(); not necessary */
622 return -1;
623 }
624 }
625#else
626 if (serv->private_info == NULL)
627 {
628 nfs3_err(NFS3ERR_PARAM);
629 return -1;
630 }
631#endif
632
633 ret = ParseURL(url, NULL, NULL, NULL, server, NULL, dir);
634 if (ret < 0)
635 {
636 nfs3_err(NFS3ERR_URL);
637 fprintf(stderr, "nfs3_open: url %s incorrect.\n", url);
638 return -1;
639 }
640
641 fho->url = (char *)malloc(strlen(url) + 1);
642 if (fho->url == NULL)
643 {
644 fprintf(stderr, "nfs3_open: fho->url is null. Can't reserve memory\n");
645 nfs3_err(NFS3ERR_MEMORY);
646 return -1;
647 }
648
649 strcpy(fho->url, url);
650
651 fh_aux = (struct nfi_nfs3_fhandle *)malloc(sizeof(struct nfi_nfs3_fhandle));
652 if (fh_aux == NULL)
653 {
654 nfs3_err(NFS3ERR_MEMORY);
655 fprintf(stderr, "nfs3_open: fh_aux is null. Can't reserve memory\n");
656 free(fho->url);
657 return -1;
658 }
659
660 server_aux = (struct nfi_nfs3_server *)serv->private_info;
661
662 getDirWithURL(serv->url, dir);
663
664 /* obtaine the NFS file handle */
665 ret = nfs3_lookup(&server_aux->fh,
666 dir,
667 &fh_aux->fh,
668 NULL,
669 server_aux->cl);
670
671 if (ret < 0)
672 {
673 nfs3_err(-ret);
674 // fprintf(stderr,"nfs3_open: Fail lookup %s in server %s (err:%d).\n",dir,serv->server,ret);
675 // free(fho->url);
676 // free(fh_aux);
677 // free(server_aux);
678 return -1;
679 }
680 nfs3_err(-ret);
681 switch (ret)
682 {
683 case NF3REG:
684 fho->type = NFIFILE;
685 break;
686 case NF3DIR:
687 fho->type = NFIDIR;
688 break;
689 }
690 fho->server = NULL;
691 fho->priv_fh = NULL;
692 fho->server = serv;
693 fho->priv_fh = (void *)fh_aux;
694
695 return 0;
696}
