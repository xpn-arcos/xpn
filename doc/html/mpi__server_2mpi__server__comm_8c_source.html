41int mpi_server_comm_init(mpi_server_param_st *params)
42{
43 int ret, provided, claimed;
44 char serv_name[HOST_NAME_MAX];
45
46 // Get timestap
47 struct timeval t0;
48 TIME_MISC_Timer(&t0);
49
50 DEBUG_BEGIN();
51
52 // MPI_Init
53
54 // Threads disable
55 if (!params->thread_mode)
56 {
57 ret = MPI_Init(&(params->argc), &(params->argv));
58 if (MPI_SUCCESS != ret)
59 {
60 debug_error("Server[%d]: MPI_Init fails :-(", -1);
61 return -1;
62 }
63 }
64 // Threads enable
65 else
66 {
67 ret = MPI_Init_thread(&(params->argc), &(params->argv), MPI_THREAD_MULTIPLE, &provided);
68 if (MPI_SUCCESS != ret)
69 {
70 debug_error("Server[%d]: MPI_Init_thread fails :-(", -1);
71 return -1;
72 }
73
74 MPI_Query_thread(&claimed);
75 if (claimed != MPI_THREAD_MULTIPLE)
76 {
77 printf("MPI_Init_thread: your MPI implementation seem not supporting thereads\n");
78 }
79 }
80
81 // params->rank = comm_rank()
82 ret = MPI_Comm_rank(MPI_COMM_WORLD, &(params->rank));
83 if (MPI_SUCCESS != ret)
84 {
85 debug_error("Server[%d]: MPI_Comm_rank fails :-(", params->rank);
86 return -1;
87 }
88
89 // params->size = comm_size()
90 ret = MPI_Comm_size(MPI_COMM_WORLD, &(params->size));
91 if (MPI_SUCCESS != ret)
92 {
93 debug_error("Server[%d]: MPI_Comm_size fails :-(", params->rank);
94 return -1;
95 }
96
97 // Open server port...
98 ret = MPI_Open_port(MPI_INFO_NULL, params->port_name);
99 if (MPI_SUCCESS != ret)
100 {
101 debug_error("Server[%d]: MPI_Open_port fails :-(", params->rank);
102 return -1;
103 }
104
105 // Generate DNS file
106 int version_len;
107 char version[MPI_MAX_LIBRARY_VERSION_STRING];
108 MPI_Get_library_version(version, &version_len);
109
110 if (strncasecmp(version, "Open MPI", strlen("Open MPI")) != 0)
111 {
112 for (int j = 0; j < params->size; j++)
113 {
114 if (j == params->rank)
115 {
116 char *ip;
117
118 ns_get_hostname(params->srv_name);
119
120 ip = ns_get_host_ip();
121 if (ip == NULL)
122 {
123 debug_error("Server[%d]: NS_PUBLISH fails :-(", params->rank);
124 return -1;
125 }
126
127 ret = ns_publish(params->dns_file, "mpi_server", params->srv_name, ip, params->port_name);
128 if (ret < 0)
129 {
130 debug_error("Server[%d]: NS_PUBLISH fails :-(", params->rank);
131 return -1;
132 }
133 }
134
135 MPI_Barrier(MPI_COMM_WORLD);
136 }
137 }
138 else
139 {
140 // Publish port name
141 MPI_Info info;
142 MPI_Info_create(&info);
143 MPI_Info_set(info, "ompi_global_scope", "true");
144
145 // struct hostent *serv_entry;
146 gethostname(serv_name, HOST_NAME_MAX); // get hostname
147 // serv_entry = gethostbyname(serv_name); // find host information
148 sprintf(params->srv_name, "%s", serv_name);
149
150 ret = MPI_Publish_name(params->srv_name, info, params->port_name);
151 if (MPI_SUCCESS != ret)
152 {
153 debug_error("Server[%d]: MPI_Publish_name fails :-(", params->rank);
154 return -1;
155 }
156 }
157
158 // Print server init information
159 MPI_Barrier(MPI_COMM_WORLD);
160
161 struct timeval t1;
162 struct timeval tf;
163 float time;
164 TIME_MISC_Timer(&t1);
165 TIME_MISC_DiffTime(&t0, &t1, &tf);
166 time = TIME_MISC_TimevaltoFloat(&tf);
167
168 if (params->rank == 0)
169 {
170 printf("\n\n");
171 printf("Time to inizialize all servers: %f s\n", time);
172 printf("\n");
173 printf("---------------------------\n");
174 printf("All XPN MPI servers running\n");
175 printf("---------------------------\n");
176 printf("\n\n");
177 }
178
179 debug_info("[SERV-COMM] server %d available at %s\n", params->rank, params->port_name);
180 debug_info("[SERV-COMM] server %d accepting...\n", params->rank);
181
182 DEBUG_END();
183
184 // Return OK
185 return 1;
186}

188int mpi_server_comm_destroy(mpi_server_param_st *params)
189{
190 int ret;
191
192 DEBUG_BEGIN();
193
194 // Close port
195 MPI_Close_port(params->port_name);
196
197 for (int i = 0; i < params->size; ++i)
198 {
199 if (params->rank == i)
200 {
201
202 int version_len;
203 char version[MPI_MAX_LIBRARY_VERSION_STRING];
204 MPI_Get_library_version(version, &version_len);
205
206 if (strncasecmp(version, "Open MPI", strlen("Open MPI")) != 0)
207 {
208 char hostname[1024];
209 ns_get_hostname(hostname);
210
211 // Unpublish port name
212 ret = ns_unpublish(params->dns_file, "mpi_server", hostname);
213 if (ret < 0)
214 {
215 debug_error("Server[%d]: ns_unpublish fails :-(", params->rank);
216 return -1;
217 }
218 }
219 else
220 {
221 // Unpublish port name
222 ret = MPI_Unpublish_name(params->srv_name, MPI_INFO_NULL, params->port_name);
223 if (MPI_SUCCESS != ret)
224 {
225 debug_error("Server[%d]: port unregistration fails :-(\n", params->rank);
226 return -1;
227 }
228 }
229 }
230
231 MPI_Barrier(MPI_COMM_WORLD);
232 }
233
234 // Finalize
235 ret = MPI_Finalize();
236 if (MPI_SUCCESS != ret)
237 {
238 debug_error("Server[%d]: MPI_Finalize fails :-(", params->rank);
239 return -1;
240 }
241
242 DEBUG_END();
243
244 // Return OK
245 return 1;
246}

290ssize_t mpi_server_comm_read_operation(mpi_server_param_st *params, MPI_Comm fd, char *data, ssize_t size, int *rank_client_id)
291{
292 int ret;
293 MPI_Status status;
294
295 DEBUG_BEGIN();
296
297 // Check params
298 if (NULL == params)
299 {
300 debug_warning("Server[%d]: ERROR: NULL arguments", -1);
301 return -1;
302 }
303 if (size == 0)
304 {
305 return 0;
306 }
307 if (size < 0)
308 {
309 debug_warning("Server[%d]: ERROR: size < 0", params->rank);
310 return -1;
311 }
312
313 // Get message
314 ret = MPI_Recv(data, size, MPI_INT, MPI_ANY_SOURCE, 0, fd, &status);
315 if (MPI_SUCCESS != ret)
316 {
317 debug_warning("Server[%d]: MPI_Recv fails :-(", params->rank);
318 }
319
320 *rank_client_id = status.MPI_SOURCE;
321
322 debug_info("[SERV-COMM] MPI SOURCE %d, MPI_TAG %d, MPI_ERROR %d\n", status.MPI_SOURCE, status.MPI_TAG, status.MPI_ERROR);
323
324 DEBUG_END();
325
326 // Return bytes read
327 return size;
328}
